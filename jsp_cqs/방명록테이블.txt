create sequence seq_visit_idx;

create table visit(
	idx Number(3) primary key,
	name Varchar2(50), -- 작성자
	content Varchar2(2000), -- 내용
	pwd Varchar2(100), -- 비밀번호
	ip Varchar2(30), -- ip주소
	regdate Date -- 작성날짜
);
